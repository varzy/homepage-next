```bash
root = true

[*]
charset = utf-8
end_of_line = lf
indent_style = space
indent_size = 2
tab_width = 2
trim_trailing_whitespace = true
insert_final_newline = true

[*.md]
trim_trailing_whitespace = false

[*.{yml,yaml}]
indent_size = 2

[*.{php,py}]
indent_size = 4
tab_width = 4

[*.blade.php]
indent_size = 2
tab_width = 2
```
