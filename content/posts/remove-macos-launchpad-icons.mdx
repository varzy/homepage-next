```bash
# 进入启动台数据库所在的文件夹。使用这个命令有可能会在控制台输出某些目录没有权限，无视即可
cd /private/var/folders/**/com.apple.dock.launchpad/db

# 执行删除并重启 Dock。这里需要把 GarageBand 换成需要删除的软件名称
sqlite3 db "delete from apps where title='GarageBand';" && killall Dock
```
